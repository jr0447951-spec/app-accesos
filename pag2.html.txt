<button id="copiar">Copiar Mensaje</button>
<button id="whatsapp">Enviar por WhatsApp</button>

<script>
/* === Obtener parámetros desde index.html === */
const params = new URLSearchParams(window.location.search);

const siteid = params.get('siteid') || '';
const sitio = params.get('sitio') || '';
const actividad = params.get('actividad') || '';
const personalRaw = params.get('personal') || '';

const personal = personalRaw
  ? personalRaw.split('|').join('\n')
  : '';

const textarea = document.getElementById('mensaje');
const tipoSelect = document.getElementById('tipoReporte');

/* === Función generadora del mensaje === */
function generarMensaje() {
  const tipo = tipoSelect.value;

  return `Buen día
Reportamos ${tipo} al sitio

${siteid} - ${sitio}

Personal:
${personal}

Empresa:
CELECT SA DE SV

Área de Movistar Responsable: EEBB

Actividad a Realizar: ${actividad}`;
}

/* === Inicializar mensaje === */
textarea.value = generarMensaje();

/* === Cambiar texto al cambiar Entrada / Salida === */
tipoSelect.addEventListener('change', () => {
  textarea.value = generarMensaje();
});

/* === Copiar al portapapeles === */
document.getElementById('copiar').addEventListener('click', () => {
  navigator.clipboard.writeText(textarea.value)
    .then(() => alert('Mensaje copiado al portapapeles'));
});

/* === Enviar por WhatsApp === */
document.getElementById('whatsapp').addEventListener('click', () => {
  const url = `https://wa.me/?text=${encodeURIComponent(textarea.value)}`;
  window.open(url, '_blank');
});
</script>
